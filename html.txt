    </div>
    <button onclick="convertScore()" class="convert-btn">Convert</button>
    <div id="result" class="result"></div>
  </div>

  <script>
    function convertScore() {
      const scoreOutOf30 = parseFloat(document.getElementById('scoreOutOf30').value);

      const scoreOutOf20 = (scoreOutOf30 / 30) * 20;
      const result = `Score out of 20: ${scoreOutOf20.toFixed(2)}`;

      document.getElementById('result').textContent = result;
    }
  </script>
